111 {
112  ASSERT(cmp_func != NULL);
113  ASSERT(hash_func != NULL);
114  ASSERT(size > 0);
115 
116  hashtable* table = MALLOC(sizeof(*table));
117  if (table) {
118  table->size = dict_prime_geq(size);

161 dict_insert_result
162 hashtable_insert(hashtable* table, void* key)
163 {
164  if (LOADFACTOR_DENOMINATOR * table->count >= LOADFACTOR_NUMERATOR * table->size) {
165  /* Load factor too high. */
166  hashtable_resize(table, table->size + 1);
167  }
168 
169  const unsigned hash = table->hash_func(key);
170  const unsigned mhash = hash % table->size;
171  hash_node* node = table->table[mhash];
172  hash_node* prev = NULL;
173  while (node && hash >= node->hash) {
174  if (hash == node->hash && table->cmp_func(key, node->key) == 0)
175  return (dict_insert_result) { &node->datum, false };
176  prev = node;
177  node = node->next;
178  }
179 
180  hash_node* add = MALLOC(sizeof(*add));
181  if (!add)
182  return (dict_insert_result) { NULL, false };
183 
184  add->key = key;
185  add->datum = NULL;
186  add->hash = hash;
187  add->prev = prev;
188  if (prev)
189  prev->next = add;
190  else
191  table->table[mhash] = add;
192  add->next = node;
193  if (node)
194  node->prev = add;
195 
196  table->count++;
197  return (dict_insert_result) { &add->datum, true };
198 }

202 {
203  const unsigned hash = table->hash_func(key);
204  hash_node* node = table->table[hash % table->size];
205  while (node && hash >= node->hash) {
206  if (hash == node->hash && table->cmp_func(key, node->key) == 0)
207  return &node->datum;
208  node = node->next;
209  }

213 static void
214 remove_node(hashtable* table, hash_node* node, unsigned mhash)
215 {
216  if (node->prev)
217  node->prev->next = node->next;
218  else
219  table->table[mhash] = node->next;
220 
221  if (node->next)
222  node->next->prev = node->prev;
223 
224  FREE(node);
225  table->count--;
226 }

229 hashtable_remove(hashtable* table, const void* key)
230 {
231  const unsigned hash = table->hash_func(key);
232  const unsigned mhash = hash % table->size;
233 
234  hash_node* node = table->table[mhash];
235  while (node && hash >= node->hash) {
236  if (hash == node->hash && table->cmp_func(key, node->key) == 0) {
237  dict_remove_result result = { node->key, node->datum, true };
238  remove_node(table, node, mhash);
239  return result;
240  }
241  node = node->next;
242  }
243  return (dict_remove_result) { NULL, NULL, false };

247 hashtable_clear(hashtable* table, dict_delete_func delete_func)
248 {
249  for (unsigned slot = 0; slot < table->size; slot++) {
250  hash_node* node = table->table[slot];
251  while (node != NULL) {
252  hash_node* next = node->next;
253  if (delete_func)
254  delete_func(node->key, node->datum);
255  FREE(node);
256  node = next;
257  }
258  table->table[slot] = NULL;
259  }
260 
261  const size_t count = table->count;
262  table->count = 0;
263  return count;

301 bool
302 hashtable_resize(hashtable* table, unsigned new_size)
303 {
304  ASSERT(new_size > 0);
305 
306  new_size = dict_prime_geq(new_size);
307  if (table->size == new_size)
308  return true;
309 
310  /* TODO: investigate whether using realloc would be advantageous. */
311  hash_node** ntable = MALLOC(new_size * sizeof(hash_node*));
312  if (!ntable)
313  return false;
314  memset(ntable, 0, new_size * sizeof(hash_node*));
315 
316  for (unsigned i = 0; i < table->size; i++) {
317  for (hash_node* node = table->table[i]; node;) {
318  hash_node* const next = node->next;
319  const unsigned mhash = node->hash % new_size;
320 
321  hash_node* search = ntable[mhash];
322  hash_node* prev = NULL;
323  while (search && node->hash >= search->hash) {
324  prev = search;
325  search = search->next;
326  }
327  if ((node->next = search) != NULL)
328  search->prev = node;
329  if ((node->prev = prev) != NULL)
330  prev->next = node;
331  else
332  ntable[mhash] = node;
333 
334  node = next;
335  }
336  }
337 
338  FREE(table->table);
339  table->table = ntable;
340  table->size = new_size;
341  return true;
342 }

344 bool
345 hashtable_verify(const hashtable* table)
346 {
347  for (unsigned slot = 0; slot < table->size; ++slot) {
348  for (hash_node* n = table->table[slot]; n; n = n->next) {
349  if (n == table->table[slot]) {
350  VERIFY(n->prev == NULL);
351  } else {
352  VERIFY(n->prev != NULL);
353  VERIFY(n->prev->next == n);
354  }
355  if (n->next) {
356  VERIFY(n->next->prev == n);
357  VERIFY(n->hash <= n->next->hash);
358  }
359  VERIFY(n->hash % table->size == slot);
360  }
361  }
362  return true;
363 }
